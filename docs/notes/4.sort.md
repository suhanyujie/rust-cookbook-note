## 插入排序

```rust
/// 插入排序
pub fn insert(arr: &mut Vec<i32>) {
    let arr_l = arr.len();
    for i in 1..arr_l {
        let mut j = i;
        while j >= 1 {
            if arr[j] < arr[j - 1] {
                let tmp = arr[j - 1];
                arr[j - 1] = arr[j];
                arr[j] = tmp;
            } else {
                break;
            }
            j -= 1;
        }
    }
}
```
